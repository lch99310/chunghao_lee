{{/* Supplement Open Graph / Twitter Card metadata so social previews (Messenger, LINE, Facebook, etc.) show a thumbnail for posts that declare a `thumbnail` front-matter image but no feature/cover image or `images` param. Hugo's internal opengraph.html already handles the latter cases, so we only fill the gap. */}} {{ if .IsPage }} {{- $images := .Resources.ByType "image" -}} {{- $hasInternal := false -}} {{- if .Params.images -}}{{ $hasInternal = true }}{{ end -}} {{- if not $hasInternal -}} {{- with $images.GetMatch "*feature*" -}}{{ $hasInternal = true }}{{- end -}} {{- end -}} {{- if not $hasInternal -}} {{- with $images.GetMatch "*cover*" -}}{{ $hasInternal = true }}{{- end -}} {{- end -}} {{- if not $hasInternal -}} {{- $thumbName := .Params.thumbnail | default "*thumb*" -}} {{- with $images.GetMatch $thumbName }} {{- end }} {{- end -}} {{ end }}
