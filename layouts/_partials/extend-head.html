{{/* Emit og:image / twitter:image with a Hugo-resized derivative of the post's hero image. Many of this site's posts ship a multi-megabyte source image (e.g. 20 MB pic.jpg); social platforms (LINE, Messenger, iMessage) reject oversized og:image URLs and silently fall back to , which on this site is a signature graphic — that's why thumbnails were wrong. Resizing to 1200x630 keeps the file under a megabyte and matches the standard social-card aspect ratio. Resolution order (first match wins): 1. .Params.images (explicit array) 2. *feature*.{jpg,png,...} (Congo theme convention) 3. *cover*.{jpg,png,...} (Congo theme convention) 4. .Params.thumbnail (this site's convention) 5. *thumb*.{jpg,png,...} 6. first image in bundle (last-resort fallback) Uses newScratch so variable assignment survives `with` block scope. */}} {{- if .IsPage -}} {{- $images := .Resources.ByType "image" -}} {{- $sc := newScratch -}} {{- $sc.Set "img" nil -}} {{- with .Params.images -}} {{- with $.Resources.GetMatch (index . 0) -}} {{- $sc.Set "img" . -}} {{- end -}} {{- end -}} {{- if not ($sc.Get "img") -}} {{- with $images.GetMatch "*feature*" -}}{{- $sc.Set "img" . -}}{{- end -}} {{- end -}} {{- if not ($sc.Get "img") -}} {{- with $images.GetMatch "*cover*" -}}{{- $sc.Set "img" . -}}{{- end -}} {{- end -}} {{- if not ($sc.Get "img") -}} {{- with .Params.thumbnail -}} {{- with $images.GetMatch . -}}{{- $sc.Set "img" . -}}{{- end -}} {{- end -}} {{- end -}} {{- if not ($sc.Get "img") -}} {{- with $images.GetMatch "*thumb*" -}}{{- $sc.Set "img" . -}}{{- end -}} {{- end -}} {{- if not ($sc.Get "img") -}} {{- with $images -}}{{- $sc.Set "img" (index . 0) -}}{{- end -}} {{- end -}} {{- with $sc.Get "img" -}} {{- $resized := . -}} {{- if ne .MediaType.SubType "svg" -}} {{- $resized = .Fill "1200x630 smart webp q82" -}} {{- end }} {{- end -}} {{- end -}}
